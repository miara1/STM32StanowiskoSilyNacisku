28void receive_data_frame(uint16_t length) {
29 // Sprawdzamy, czy ramka jest wystarczająco długa oraz czy zaczyna się od 'U'
30 if (length < 9 || odbiorUART.buffer[0] != NAGLOWEK_RAMKI_APLIKACJI) { // Zakładamy minimalną długość: U + dane + 4 znaki CRC + \r\n
31 printf("Błąd: Niepoprawna ramka lub brak nagłówka 'U'\n");
32 return;
33 }
34 //Zatrzymanie przerwan wsyzstkich timerow
35 HAL_TIM_Base_Stop_IT(&htim2);
36 HAL_TIM_Base_Stop_IT(&htim3);
37 HAL_TIM_Base_Stop_IT(&htim4);
38 //Czyszczenie buforow
39// memset(tablicaPomiarowDynamicznych, 0, sizeof(tablicaPomiarowDynamicznych));
40// memset(tablicaHistoriiPrzedPomiarem, 0, sizeof(tablicaHistoriiPrzedPomiarem));
41
42 // Oddzielamy dane od CRC (ostatnie 4 znaki przed '\r\n' to tekstowe CRC)
43 char crc_text[5] = {0};
44 memcpy(crc_text, &odbiorUART.buffer[length - 6], 4); // Kopiujemy 4 znaki tekstowego CRC (np. "a9f9")
45
46 // Parsujemy tekstowe CRC na wartość liczbową
47 uint16_t crc_received = parse_crc_from_text(crc_text);
48 //printf("Mam nowe crc_received: %d\r\n", crc_received);
49
50 // Obliczamy CRC dla odebranych danych (bez 'U' i bez CRC)
51 uint16_t crc_calculated = crc16_ccitt_false((const char*)odbiorUART.buffer, length - 6);
52 //printf("A crc calculated t o : %d\r\n", crc_calculated);
53
54 // Sprawdzamy poprawność CRC
55 if (crc_calculated == crc_received) {
56 // Przekazujemy dane bez CRC i nagłówka 'U' do przetworzenia
57 process_received_data(odbiorUART.buffer + 2, length - 7); // -7 to U + CRC (4 znaki) + \r\n
58 } else {
59 printf("Błąd CRC! Odebrano: %u, Obliczono: %u\n", crc_received, crc_calculated);
60 }
61}

70void sendDataRamkaDynamT(uint32_t *tablicaPomiarowDynamicznych, int czasPomiaruDynam)
71{
72 const int maxBufSize = 3*MAX_POMIARY; // Większy bufor, aby zmieścić całą ramkę
73 char frame[maxBufSize]; // Bufor dla całej ramki
74 int czasPomNa2 = czasPomiaruDynam/2;
75 int index = 0;
76
77 // Dodanie nagłówka "P D [C5 lub C6]"
78 index += snprintf(frame + index, sizeof(frame) - index, "P D %s ", warunkiPomiaru.idCzujnika);
79
80 // Dodanie informacji o rodzaju pomiaru: "T [czasPomiaruDynam]"
81 index += snprintf(frame + index, sizeof(frame) - index, "T %d ", czasPomiaruDynam);
82
83
84 int wyliczonyIdxHistorii = warunkiPomiaru.indeksHistorii;
85 // Dodanie historii pomiaru
86 for ( int licznik = 0; licznik<ROZM_HISTORIA; licznik++)
87 {
88 index += snprintf(frame + index, sizeof(frame) - index, "%lu ", danePomiaroweDynam.tablicaHistoriiPrzedPomiarem[wyliczonyIdxHistorii]);
89 wyliczonyIdxHistorii = (wyliczonyIdxHistorii + 1) % ROZM_HISTORIA;
90 }
91 // Dodanie danych pomiarowych
92
93 for (int i = 0; i < czasPomNa2; i++) {
94 if(danePomiaroweDynam.tablicaPomiarowDynamicznych[i] > 0)
95 {
96 index += snprintf(frame + index, sizeof(frame) - index, "%lu ", tablicaPomiarowDynamicznych[i] );
97 }
98 }
99
100 // Usuwanie ostatniej spacji, jeśli istnieje
101 if (index > 0 && frame[index - 1] == ' ') {
102 index--; // Cofnięcie indeksu, aby pominąć końcową spację
103 }
104
105 // Obliczenie CRC dla danych w ramce (bez końcowej spacji)
106 uint16_t crc = crc16_ccitt_false((char *)frame, index);
107
108 // Dodanie CRC na końcu ramki
109 index += snprintf(frame + index, sizeof(frame) - index, " %u\r\n", crc);
110
111 // Wysyłanie całej ramki jednorazowo
112 HAL_UART_Transmit(&huart2, (uint8_t *)frame, index, HAL_MAX_DELAY);
113
114 // Wyczyszczenie tablicy po wysłaniu danych
115 memset(tablicaPomiarowDynamicznych, 0, sizeof(int) * MAX_POMIARY); // Zerowanie zawartości tablicy
116}

124void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
125 if (huart->Instance == USART2) {
126 static uint16_t length = 0;
127
128 // Sprawdzamy, czy ramka zaczyna się od 'U'
129 if (length == 0 && odbiorUART.receivedData != NAGLOWEK_RAMKI_APLIKACJI) {
130 printf("Nie ma U na poczatku\r\n");
131 // Ignorujemy, jeśli pierwszy znak nie jest 'U'
132 HAL_UART_Receive_IT(&huart2, &odbiorUART.receivedData, 1);
133 return;
134 }
135
136 odbiorUART.buffer[length++] = odbiorUART.receivedData; // Buforowanie odebranego znaku
137
138 // Sprawdzamy, czy ramka kończy się na '\r\n'
139 if (length >= 2 && odbiorUART.buffer[length - 2] == '\r' && odbiorUART.buffer[length - 1] == '\n') {
140 receive_data_frame(length); // Przetwarzamy ramkę danych
141 length = 0; // Resetujemy długość bufora po zakończeniu ramki
142 }
143
144 // Włączamy ponownie przerwanie odbioru danych UART
145 HAL_UART_Receive_IT(&huart2, &odbiorUART.receivedData, 1);
146 }
